module mem #(
parameter ADDR_WIDTH = 10, // Number of address bits
parameter DATA_WIDTH = 16 // Data width
) (
input wire i_clk,
input wire i_rst,
input wire [ADDR_WIDTH-1:0] i_addr,
input wire i_wr_en, // Write enable
input wire i_rd_en, // Read enable
input wire [DATA_WIDTH-1:0] i_wr_data, // Data input
output reg [DATA_WIDTH-1:0] o_rd_data // Data output
);
// Memory declaration
reg [DATA_WIDTH-1:0] mem [(2**ADDR_WIDTH)];
always_ff @(posedge i_clk) begin
// Write to memory
if (i_wr_en)
mem[i_addr] <= i_wr_data;
end
always_ff @(posedge i_clk) begin
// Read from memory
if (i_rst) begin
o_rd_data <= 0;
end else begin
if (i_rd_en)
o_rd_data <= mem[i_addr];
end
end
initial begin
$readmemh("init.mem", mem);
end
endmodule
